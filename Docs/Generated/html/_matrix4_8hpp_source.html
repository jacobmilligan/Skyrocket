137  Matrix4<T> rotate(const T angle, const Vector3<T>& axis_vec)
138  {
139  auto axis = axis_vec;
140  auto rotation = *this;
141 
142  axis.normalize();
143 
144  auto dsin = static_cast<T>(sin(angle));
145  auto dcos = static_cast<T>(cos(angle));
146  auto ocos = static_cast<T>(1.0 - dcos);
147 
148 
149  // Rodrigues' Rotation Formula
150  rotation[0][0] = dcos + (axis.x * axis.x) * ocos;
151  rotation[0][1] = axis.y * axis.x + axis.z * dsin;
152  rotation[0][2] = axis.z * axis.x * ocos - axis.y * dsin;
153 
154  rotation[1][0] = axis.x * axis.y * ocos - axis.z * dsin;
155  rotation[1][1] = dcos + (axis.y * axis.y) * ocos;
156  rotation[1][2] = axis.z * axis.y * ocos + axis.x * dsin;
157 
158  rotation[2][0] = axis.x * axis.z * ocos + axis.y * dsin;
159  rotation[2][1] = axis.y * axis.z * ocos - axis.x * dsin;
160  rotation[2][2] = dcos + (axis.z * axis.z) * ocos;
161 
162  return rotation;
163  }

172  Matrix4<T> ortho(const T left, const T right, const T bottom,
173  const T top)
174  {
175  Matrix4<T> result;
176 
177  // Width and height
178  result[0][0] = static_cast<T>(2) / (right - left);
179  result[1][1] = static_cast<T>(2) / (top - bottom);
180 
181  // OpenGL expects a lhs for the perspective divide stage
182  // so the z coordinate needs to be negated to flip it
183  result[2][2] = static_cast<T>(-1);
184 
185  result[3][0] = -(right + left) / (right - left);
186  result[3][1] = -(top + bottom) / (top - bottom);
187 
188  return result;
189  }

200  Matrix4<T> ortho(const T left, const T right, const T bottom,
201  const T top, const T near, const T far)
202  {
203  Matrix4<T> result;
204 
205  // Width and height
206  result[0][0] = static_cast<T>(2) / (right - left);
207  result[1][1] = static_cast<T>(2) / (top - bottom);
208 
209  result[3][0] = -(right + left) / (right - left);
210  result[3][1] = -(top + bottom) / (top - bottom);
211 
212  // OpenGL expects a lhs for the perspective divide stage
213  // so the z coordinate needs to be negated to flip it
214  result[2][2] = static_cast<T>(-2) / (far - near); // -2 to work with gl instead of using -1
215  result[3][2] = -(far + near) / (far - near); // f + n instead of just n in numer for gl compatibility
216 
217  return result;
218  }

226  Matrix4<T> look_at(const Vector3<T>& eye, const Vector3<T>& target,
227  const Vector3<T>& up)
228  {
229  Matrix4<T> result;
230 
231  Vector3<T> f(target - eye); // towards center
232  f.normalize();
233 
234  auto s = f.cross(up);
235  s.normalize();
236 
237  auto u = s.cross(f); // pointing up
238 
239  // Up
240  result[0][1] = u.x;
241  result[1][1] = u.y;
242  result[2][1] = u.z;
243 
244  // Right
245  result[0][0] = s.x;
246  result[1][0] = s.y;
247  result[2][0] = s.z;
248 
249  // Negative target for rhs
250  result[0][2] = -f.x;
251  result[1][2] = -f.y;
252  result[2][2] = -f.z;
253 
254  // Calculate translations - always in 4th column
255  result[3][0] = -s.dot(eye); // -position.x
256  result[3][1] = -u.dot(eye); // -position.y
257  result[3][2] = f.dot(eye); // z for rhs instead of -z
258 
259  return result;
260  }
