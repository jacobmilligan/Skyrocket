31 bool MetalGDI::initialize(Viewport* viewport)
32 {
33  //---------------------------------------
34  // Initialize device and command queue
35  //---------------------------------------
36 
37  NSError* err = nil;
38  device_ = MTLCreateSystemDefaultDevice();
39 
40  set_viewport(viewport);
41 
42  command_queue_ = [device_ newCommandQueue];
43 
44  //--------------------------------
45  // Load library and shader path
46  //--------------------------------
47  Path lib_path(SKY_RESOURCE_DIRECTORY);
48  lib_path.append("Shaders/Metal/Lib.metallib");
49 
50  NSString* nspath = [NSString stringWithUTF8String:lib_path.str()];
51  library_ = [device_ newLibraryWithFile:nspath error:&err];
52 
53  if ( library_ == nil ) {
54  SKY_ERROR("Graphics Device Interface", "Couldn't load metal library with path %s: NSError: %s",
55  lib_path.str(), [[err localizedDescription] UTF8String]);
56  return false;
57  }
58 
59  //--------------------------------
60  // Load default library
61  //--------------------------------
62 
63  const char* default_src = R"(
64 #include <metal_stdlib>
65 
66 using namespace metal;
67 
68 struct Vertex {
69  float4 position [[position]];
70  float4 color;
71 };
72 
73 vertex Vertex basic_vertex( device Vertex* vertices [[buffer(0)]], uint vid [[vertex_id]] )
74 {
75  return vertices[vid];
76 }
77 
78 fragment float4 basic_fragment(Vertex in [[stage_in]])
79 {
80  return in.color;
81 }
82  )";
83 
84  NSString* nssrc = [NSString stringWithUTF8String:default_src];
85 
86  default_library_ = [device_ newLibraryWithSource:nssrc options:nil error:&err];
87  if ( default_library_ == nil ) {
88  SKY_ASSERT(default_library_ != nil, "Default Metal Library loads correctly (see NSError: %s)",
89  [[err localizedDescription] UTF8String]);
90  return false;
91  }
92 
93  default_vshader_ = [default_library_ newFunctionWithName:@"basic_vertex"];
94  default_fragshader_ = [default_library_ newFunctionWithName:@"basic_fragment"];
95 
96  //--------------------------------
97  // Load main render pipeline
98  //--------------------------------
99 
100  MTLRenderPipelineDescriptor* pipeline_descriptor = [MTLRenderPipelineDescriptor new];
101  pipeline_descriptor.colorAttachments[0].pixelFormat = MTLPixelFormatBGRA8Unorm;
102  pipeline_descriptor.vertexFunction = default_vshader_;
103  pipeline_descriptor.fragmentFunction = default_fragshader_;
104 
105  render_pipeline_ = [device_ newRenderPipelineStateWithDescriptor:pipeline_descriptor error:&err];
106 
107  if ( render_pipeline_ == nil ) {
108  SKY_ERROR("Graphics Device Interface", "Couldn't initialize main render pipeline state: NSError: %s",
109  [[err localizedDescription] UTF8String]);
110  return false;
111  }
112 
113  return true;
114 }

168 bool MetalGDI::set_shaders(const uint32_t vertex_id, const uint32_t fragment_id)
169 {
170  if ( !shaders_.contains(vertex_id) || vertex_id == shaders_.invalid_id ) {
171  SKY_ERROR("Shader", "%" PRIu32 " is an invalid vertex shader ID", vertex_id);
172  return false;
173  }
174  if ( !shaders_.contains(fragment_id) || fragment_id == shaders_.invalid_id ) {
175  SKY_ERROR("Shader", "%" PRIu32 " is an invalid vertex shader ID", fragment_id);
176  return false;
177  }
178 
179  return false;
180 }

182 void MetalGDI::present()
183 {
184  if ( mtl_layer_ == nil ) {
185  return;
186  }
187 
188  command_buffer_[prev_buf] = [command_queue_ commandBufferWithUnretainedReferences];
189 
190  id<CAMetalDrawable> drawable = [mtl_layer_ nextDrawable];
191  if ( drawable == nil ) {
192  SKY_ERROR("Renderer", "Couldn't get next CAMetalDrawable");
193  return;
194  }
195 
196  MTLRenderPassDescriptor* rpd = [MTLRenderPassDescriptor renderPassDescriptor];
197 
198  if ( rpd == nil ) {
199  SKY_ERROR("Renderer", "Couldn't create a RenderPassDescriptor");
200  return;
201  }
202 
203  rpd.colorAttachments[0].texture = drawable.texture;
204  rpd.colorAttachments[0].loadAction = MTLLoadActionClear;
205  rpd.colorAttachments[0].clearColor = MTLClearColorMake(0.0, 0.0, 0.0, 0.0);
206  rpd.colorAttachments[0].storeAction = MTLStoreActionStore;
207 
208  render_encoder_ = [command_buffer_[prev_buf] renderCommandEncoderWithDescriptor:rpd];
209 
210  [rpd release];
211 
212  [render_encoder_ setRenderPipelineState:render_pipeline_];
213 
214  process_commands();
215 
216  [render_encoder_ endEncoding];
217 
218  [command_buffer_[prev_buf] presentDrawable:drawable];
219 
220  [command_buffer_[prev_buf] commit];
221 
222  [render_encoder_ release];
223  [drawable release];
224 }
