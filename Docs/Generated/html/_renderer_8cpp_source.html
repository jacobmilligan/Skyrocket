113 void Renderer::frame()
114 {
115  active_ = true;
116  std::mutex mut;
117 
118  while ( active_ ) {
119  std::unique_lock<std::mutex> lock(mut);
120 
121  cv_.wait(lock, [this]() { return notified_; });
122  notified_ = false;
123 
124  rendering_ = true;
125  if ( gdi_ != nullptr ) {
126  gdi_->swap_buffers();
127  gdi_->present();
128  }
129  rendering_ = false;
130  }
131 }
