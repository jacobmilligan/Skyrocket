158  float distance(const Vector4<T>& target) const
159  {
160  auto ydist = y - target.y;
161  auto xdist = x - target.x;
162  auto zdist = z - target.z;
163  auto wdist = w - target.w;
164  return static_cast<float>(sqrt(
165  (ydist * ydist) + (xdist * xdist) + (zdist * zdist) + (wdist * wdist)
166  ));
167  }
